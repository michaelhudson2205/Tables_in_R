library(gt)
library(tidyverse)
library(modeldata)
scat_table1 <- scat |> 
  select(Species,
         Site,
         Length,
         Diameter,
         Mass,
         d13C,
         d15N) |> 
  group_by(Species) |> 
  summarise(across(where(is.numeric),
                   \(x) round(mean(x, na.rm = TRUE), 1)),
            count = n(), .groups = "drop")

scat_table2 <- scat |> 
  select(Species,
         Location,
         Site,
         Length,
         Diameter,
         Mass,
         d13C,
         d15N) |> 
  group_by(Species, Site) |> 
  summarise(across(where(is.numeric),
                   \(x) round(mean(x, na.rm = TRUE), 1)),
            count = n(), .groups = "drop")
